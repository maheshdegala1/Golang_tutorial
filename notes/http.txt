🔹 1️⃣ HTTP Server (net/http package)

Import:

import (
	"net/http"
)
✅ http.HandleFunc

Registers a route.

http.HandleFunc("/hello", handlerFunction)
✅ http.ListenAndServe

Starts the server.

http.ListenAndServe(":8080", nil)

With error handling:

err := http.ListenAndServe(":8080", nil)
if err != nil {
	log.Fatal(err)
}

🔹 5️⃣ JSON Encoding / Decoding (encoding/json)

Import:

import "encoding/json"
✅ Encode (Send JSON Response)

Better way:

err := json.NewEncoder(w).Encode(res)
if err != nil {
	http.Error(w, err.Error(), http.StatusInternalServerError)
}

What it does:

Converts struct → JSON

Writes to response

✅ Decode JSON (Read Request Body)
var user User

err := json.NewDecoder(r.Body).Decode(&user)
if err != nil {
	http.Error(w, err.Error(), http.StatusBadRequest)
	return
}

What it does:

Converts JSON → struct

🔹 6️⃣ Full Example (GET + POST + JSON)
package main

import (
	"encoding/json"
	"net/http"
)

type Response struct {
	Message string `json:"message"`
}

func handler(w http.ResponseWriter, r *http.Request) {

	switch r.Method {

	case http.MethodGet:
		name := r.URL.Query().Get("name")

		res := Response{
			Message: "Hello " + name,
		}

		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(res)

	case http.MethodPost:
		var req Response

		err := json.NewDecoder(r.Body).Decode(&req)
		if err != nil {
			http.Error(w, err.Error(), http.StatusBadRequest)
			return
		}

		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(req)

	default:
		http.Error(w, "Method not allowed", http.StatusMethodNotAllowed)
	}
}

func main() {
	http.HandleFunc("/", handler)
	http.ListenAndServe(":8080", nil)
}
